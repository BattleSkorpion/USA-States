	fire_only_once = no

		days = 100
		modifier = {
			factor = 2
			any_owned_state = {
				has_state_flag = SID_adequate_farming
			}
		}
		modifier = {
			factor = 2
			any_owned_state = {
				has_state_flag = SID_advanced_farming
			}
		}
	}

	trigger = {
		OR = {
			any_owned_state = {
				has_state_flag = SID_minimal_farming
			}
			NOT = {
				has_completed_focus = SID_continue_economic_investment
			}
		}
	}

	option = {
		name = Idaho.4.a
		custom_effect_tooltip = SID_drought_can_be_fixed_by_investing_in_farming_tt
		random_owned_controlled_state = {
			limit = {
				NOT = {
					has_state_flag = SID_adequate_farming
					has_state_flag = SID_advanced_farming
					state = 612
				}
			}
			add_manpower = -2500
		}
	}
}
country_event = {
	id = Idaho.5
	title = Idaho.5.t
	desc = Idaho.5.d
	#picture = GFX_

	is_triggered_only = no
	fire_only_once = no

	trigger = {
		has_completed_focus = SID_continue_economic_investment
		any_owned_state = {
			has_state_flag = SID_advanced_farming
		}
		check_variable = { SID_farming > 0.5 }
	}

	mean_time_to_happen = {
		days = 200
		modifier = {
			factor = 0.5
			check_variable = { SID_farming > 1 }
		}
		modifier = {
			factor = 0.5
			check_variable = { SID_farming > 2 }
		}
	}

	option = {
		name = Idaho.5.a
		every_controlled_state = {
			limit = {
				OR = {
					has_state_flag = SID_advanced_farming
					state = 612
				}
				
			}
			if = {
				limit = {
					check_variable = { SID_farming < 2 }
				}
				add_manpower = 5000
			}
			else = {
				add_manpower = 10000
			}
		}
	}
}
