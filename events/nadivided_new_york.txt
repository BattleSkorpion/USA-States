        days = 1
    }

            factor = 60

        hidden_effect = {

                id = new_york.2
                days = 10

                id = new_york.3
                days = 30
                random = 10
            }
        }
    }
    option = { #They have a point
        name = new_york.1.b
        ai_chance = {
            factor = 40
        }
        custom_effect_tooltip = SNY_the_army_will_gain_influence_tt
        hidden_effect = {
            set_country_flag = SNY_army_influence1
            country_event = {
                id = new_york.2
                days = 5

            country_event = {
                id = new_york.3
                days = 10

            }
        }
    }
}
