			add_idea = SMI_black_legion_democratic_1
		}

		limit = {
			has_idea = SMI_black_legion_democratic_1
		}
		swap_ideas = {
			remove_idea = SMI_black_legion_democratic_1
			add_idea = SMI_black_legion_democratic_2
		}
	}
}
