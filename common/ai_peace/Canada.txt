# Default Logic behaviour that the AI will fallback on if they are in a civil war
# Should focus on getting the nation united again no matter their ideology
# Written by Henrik "Groogy" Hansson

america_peace = {
	# ROOT = Taker, FROM = Giver
	enable = {
		AND = {
			FROM = {
				is_canadian_province_trigger = yes
			}
			ROOT = {
				is_canadian_province_trigger = yes
			}
		}
	}

	annex_randomness = 0
	liberate_randomness = 0
	puppet_randomness = 0
	take_states_randomness = 0
	force_government_randomness = 0
	
	# ROOT = Taker, FROM = Giver
	annex = {
		factor = 4000
		
		modifier = {
			factor = 1000
			AND = {
				FROM = {
					any_owned_state = {
						AND = {
						is_in_home_area = yes
							OR = {
								is_on_continent = western_canada
								is_on_continent = eastern_canada
								is_on_continent = northern_canada
							}
						}
					}
				}
				ROOT = {
					is_canadian_province_trigger = yes
				}
			}
		}
	}

	# ROOT = Taker, FROM = Liberated
	liberate = {
		factor = 0
	}

	# ROOT = Taker, FROM = Giver
	puppet = {
		factor = 0
	}

	# ROOT = Taker, FROM = Giver
	puppet_all = {
		base = 0 
	}
	
	# ROOT = STATE, FROM = Taker, FROM.FROM = Giver
	puppet_state = {
		base = 0 
	}
	
	# ROOT = Taker, FROM = State
	take_states = {
		factor = 0
	}

	# ROOT = Taker, FROM = Giver
	force_government = {
		factor = 0
	}
}
