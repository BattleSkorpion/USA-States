# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { # Culture
	name = GetCulture
	text = {
		trigger = {
			OR = {
				check_variable = { culture_gridbox_value =  4 }
				check_variable = { culture_var =  4 }
			}
		}
		localization_key = CGUI_BOZO_MODZ
	}
	text = {
		trigger = {
			OR = {
				check_variable = { culture_gridbox_value = 1 }
				check_variable = { culture_var = 1 }
			}
		}
		localization_key = CGUI_GREEK
	}
	text = {
		trigger = {
			OR = {
				check_variable = { culture_gridbox_value = 2 }
				check_variable = { culture_var = 2 }
			}
		}
		localization_key = CGUI_TURKISH
	}
	text = {
		trigger = {
			OR = {
				check_variable = { culture_gridbox_value = 3 }
				check_variable = { culture_var = 3 }
			}
		}
		localization_key = CGUI_QUEBECOIS
	}
}
